算法训练-动态数组

2020-03-11 蓝桥杯 

/*
试题 算法训练 动态数组使用
  
资源限制
时间限制：1.0s   内存限制：512.0MB
从键盘读入n个整数，使用动态数组存储所读入的整数，并计算它们的和与平均值分别输出。要求尽可能使用函数实现程序代码。平均值为小数的只保留其整数部分。
样例输入: 
5 
3 4 0 0 2
样例输出:
9  1
样例输入: 
7
3 2 7 5 2 9 1
样例输出:
29  4
*/
#include <algorithm>
#include <cstdio>
#include <vector>
using namespace std;
int getSum(vector<int> arr){
	int sum=0;
	for(int i=0;i<arr.size();i++){
		sum+=arr[i];
	}
	return sum;
} 
int main() {
    int n;
    vector<int> arr;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
		int x;
		scanf("%d",&x);
		arr.push_back(x);
	}
	
	int sum = getSum(arr);
	int avg = sum/arr.size();
	printf("%d %d",sum,avg);
    return 0;
}
